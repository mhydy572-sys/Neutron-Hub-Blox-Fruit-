                        end
                    end)
                end
                pcall(function()
                    if Fruit:FindFirstChild("Fruit") and Fruit.Fruit:FindFirstChild("AnimationController") then
                        Fruit.Fruit["AnimationController"]:LoadAnimation(Fruit.Fruit.Idle):Play()
                    end
                end)
            end
        end)
